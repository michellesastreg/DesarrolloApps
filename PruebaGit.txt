    public static int esMultiplo (int a, int b) 
    
    
        {
        
            // revisar si a es multiplo de b  


            if (a % b == 0) 

                {
                    // regresar algo 

                    return 1;
                }

            else 

                {
                   // regresar algo 

                   return 0; 
                }

        }

    public static int potencia (int base, int potencia) 
    
        {

            int resultado = 1;
        
            // correr este proceso hasta que el numero de veces sea mayor que la potencia 


            for (int i = 0; i < potencia; i++)
                
                {
                    // hacer la multiplicacion y repetir 

                    resultado = resultado * base;
                }

            // regresar algo 

            return resultado;
        

        }

    public static String[] generarTextos(String[] nombres, int [] calificaciones)

        {
           // revisar tamano de array  

            int size = nombres.length;

            String [] array = new String [size];

           // correr este proceso hasta que el numero de veces sea mayor que el tamano del array 

            for (int i = 0; i < size; i++)

            {
                // obtener datos de el index actual e imprimir

                array[i] = "La calificacion de " + nombres [i] + " es de " + calificaciones[i];
                System.out.println(array[i]);
            }

            // regresar algo 

            return array;


        }

    public static double promedioGrupo (int[] calis) 

       {
           // sum = cantidad de numeros que tengo en mi array 
           double sum = 0;

           for (int i=0; i < calis.length; i++) 

               {
                    sum = sum + calis[i];
               }


          // sacar promedio
          double promedio = sum / calis.length;
        
          // imprimir

          System.out.println("Ejercicio4: " + promedio);

          // regresa algo
          return promedio;
        
        }
